```typescript
import gtdPersonal from './gtd-personal.json';

export interface BinderTypeConfig {
  name: string;
  purpose: string;
  categoryOrdering: string[];
  supportedAtomTypes: string[];
  questionTemplates: Record<string, { question: string; options: Record<string, string[]> }>;
  backgroundCloudEnrichment: boolean;
}

// Default config — extensible by adding new JSON files
const BINDER_CONFIGS: Record<string, BinderTypeConfig> = {
  'gtd-personal': gtdPersonal as BinderTypeConfig,
};

export function getBinderConfig(type: string = 'gtd-personal'): BinderTypeConfig {
  return BINDER_CONFIGS[type] ?? BINDER_CONFIGS['gtd-personal'];
}
```

```typescript
export interface EntityGraphEntry {
  id: string;
  sourceAtomId: string;
  entityType: string;  // 'outcome' | 'deadline' | 'context' | 'reference' | 'person' | 'parent' | 'related' | 'context-tag' | 'next-action'
  entityValue: string;
  relationship: string;  // 'has-outcome' | 'has-deadline' | 'has-context' | 'has-reference' | 'involves-person' | 'parent-of' | 'child-of' | 'related-to' | 'tagged-with' | 'has-next-action'
  targetValue: string;   // Optional target (e.g., child atom ID for parent-of)
  createdAt: number;
}
```

Export functions:
- `seedEntityRelationship(entry: Omit<EntityGraphEntry, 'id' | 'createdAt'>): Promise<void>` — generates UUID id and timestamp, writes via writeQueue
- `getRelationships(atomId: string): Promise<EntityGraphEntry[]>` — queries both sourceAtomId = atomId AND targetValue = atomId (single-direction storage + bidirectional query per Research recommendation)
- `getRelationshipsByType(atomId: string, entityType: string): Promise<EntityGraphEntry[]>` — filtered by compound index

Use single-direction storage (store "parent-of" once, not "parent-of" + "child-of" as separate records). The `getRelationships` helper provides bidirectional lookup. This follows the Research recommendation for fewer records and simpler CRDT conflict resolution.
