      ```typescript
      if (momentum) {
        const config = { signalCategoryMap: signalCategoryMap ?? {}, entityCategoryMap: entityCategoryMap ?? {}, entityTypePriorityWeights: entityTypePriorityWeights ?? {} };
        const ranking = predictEnrichmentOrder(cognitiveSignals ?? null, momentum, entityScores ?? {}, depthMap ?? {}, config);
        const rankMap = new Map(ranking.map((r, i) => [r.category, i]));
        questions.sort((a, b) => (rankMap.get(a.category) ?? 999) - (rankMap.get(b.category) ?? 999));
      } else if (cognitiveSignals) {
        // Fallback: no momentum available (e.g., caller didn't provide it)
        // Use simple signal inversion as a degraded fallback — keeps existing behavior until all callers migrate
        questions.sort((a, b) => {
          const relA = computeFallbackRelevance(a.category, cognitiveSignals, signalCategoryMap);
          const relB = computeFallbackRelevance(b.category, cognitiveSignals, signalCategoryMap);
          return relB - relA;
        });
      }
      ```
    - Add a private `computeFallbackRelevance()` function as a minimal inline fallback (same logic as old computeSignalRelevance but reads from config map instead of hardcoded const). This ensures the function is not exported — it exists only as a safety net during transition.
