    **Part B: Sequence context injection in tier2-handler.ts**

    6. Add a `getSequenceContext` helper function in tier2-handler.ts that sends GET_SEQUENCE_CONTEXT to the worker and returns a Promise<number[]>:
       ```typescript
       function getSequenceContext(
         worker: Worker,
         binderId: string,
         windowSize: number = 5,
       ): Promise<number[]> {
         return new Promise((resolve) => {
           const id = crypto.randomUUID();
           const timeout = setTimeout(() => {
             worker.removeEventListener('message', handler);
             resolve(new Array(128).fill(0)); // timeout fallback = zero-pad
           }, 2000);

           const handler = (event: MessageEvent) => {
             const msg = event.data;
             if (msg.id !== id) return;
             worker.removeEventListener('message', handler);
             clearTimeout(timeout);
             if (msg.type === 'SEQUENCE_CONTEXT_RESULT') {
               resolve(msg.context);
             } else {
               resolve(new Array(128).fill(0)); // error fallback = zero-pad
             }
           };

           worker.addEventListener('message', handler);
           worker.postMessage({ type: 'GET_SEQUENCE_CONTEXT', id, binderId, windowSize });
         });
       }
       ```
